API 文档 - 用户管理

注册用户

URL: /user/register

方法: POST

请求体:

{
    "username": "string",
    "password": "string",
    "email": "string"
}

响应:

{
    "success": true,
    "data": {
        "userId": "string",
        "username": "string",
        "email": "string"
    },
    "message": "注册成功"
}

用户登录

URL: /user/login

方法: POST

请求体:

{
    "username": "string",
    "password": "string"
}

响应:

{
    "success": true,
    "data": {
        "userId": "string",
        "username": "string",
        "token": "string"
    },
    "message": "登录成功"
}

获取用户信息

URL: /user/info

方法: POST

请求参数:

{
    "with_token": "string"
}

响应:

{
    "success": true,
    "data": {
        "userId": "string",
        "username": "string",
        "email": "string",
        "token": "string"
    },
    "message": "获取用户信息成功"
}
